# EraseA12 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native macOS app (Swift + AppKit) that detects PWND DFU iOS devices and performs obliteration erase, replacing the existing Python/PyQt5 usbobliter8 tool.

**Architecture:** Layered AppKit app with IOKit device monitoring, libirecovery C library bridge, plist-driven device config, and a 4-step guided wizard UI with Liquid Glass visual style. Vendor static libs compiled from source with minos 10.15 for full version range support.

**Tech Stack:** Swift 5.8, AppKit, IOKit, libirecovery (C, static linked), xcodegen for project generation, Xcode 14.3

- [ ] **Step 1: Write build-vendor-libs.sh**

```bash
#!/bin/bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
VENDOR_DIR="$PROJECT_DIR/Vendor/libirecovery"
BUILD_DIR="$PROJECT_DIR/.vendor-build"
PREFIX_X86="$BUILD_DIR/prefix-x86"
PREFIX_ARM="$BUILD_DIR/prefix-arm"

DEPLOY_TARGET="10.15"
NPROC=$(sysctl -n hw.ncpu)

# Source tarballs — download if not cached
LIBPLIST_VER="2.7.0"
LIBUSB_VER="1.0.30"
GLUE_VER="1.3.2"
IRECV_VER="1.3.1"

LIBPLIST_URL="https://github.com/libimobiledevice/libplist/releases/download/${LIBPLIST_VER}/libplist-${LIBPLIST_VER}.tar.bz2"
LIBUSB_URL="https://github.com/libusb/libusb/releases/download/v${LIBUSB_VER}/libusb-${LIBUSB_VER}.tar.bz2"
GLUE_URL="https://github.com/libimobiledevice/libimobiledevice-glue/releases/download/${GLUE_VER}/libimobiledevice-glue-${GLUE_VER}.tar.bz2"
IRECV_URL="https://github.com/libimobiledevice/libirecovery/releases/download/${IRECV_VER}/libirecovery-${IRECV_VER}.tar.bz2"

echo "=== EraseA12 Vendor Lib Builder ==="
echo "Building universal static libs with MACOSX_DEPLOYMENT_TARGET=${DEPLOY_TARGET}"
echo ""

rm -rf "$BUILD_DIR"
mkdir -p "$BUILD_DIR" "$BUILD_DIR/tarballs" "$PREFIX_X86" "$PREFIX_ARM"

download() {
    local url="$1"
    local name="$(basename "$url")"
    local dest="$BUILD_DIR/tarballs/$name"
    if [ ! -f "$dest" ]; then
        echo "Downloading $name..."
        curl -sL "$url" -o "$dest"
    fi
    echo "Extracting $name..."
    tar xf "$dest" -C "$BUILD_DIR"
}

download "$LIBPLIST_URL"
download "$LIBUSB_URL"
download "$GLUE_URL"
download "$IRECV_URL"

build_lib() {
    local src_dir="$1"
    local prefix="$2"
    local arch="$3"
    local extra_args="${4:-}"

    echo "--- Building $(basename "$src_dir") for $arch ---"

    local cc_flags=""
    local ld_flags=""
    if [ "$arch" = "x86_64" ]; then
        cc_flags="-target x86_64-apple-macos${DEPLOY_TARGET}"
        ld_flags="-target x86_64-apple-macos${DEPLOY_TARGET}"
    else
        cc_flags="-target arm64-apple-macos${DEPLOY_TARGET}"
        ld_flags="-target arm64-apple-macos${DEPLOY_TARGET}"
    fi

    cd "$src_dir"
    make clean 2>/dev/null || true

    CFLAGS="$cc_flags -mmacosx-version-min=${DEPLOY_TARGET} -O2" \
    LDFLAGS="$ld_flags -mmacosx-version-min=${DEPLOY_TARGET}" \
    ./configure \
        --host="$arch-apple-darwin" \
        --prefix="$prefix" \
        --enable-static \
        --disable-shared \
        $extra_args

    make -j"$NPROC"
    make install
}

# Build order: libplist → libusb → libimobiledevice-glue → libirecovery
# Each depends on the previous ones

for ARCH in x86_64 arm64; do
    echo ""
    echo "========== Building for $ARCH =========="
    PREFIX="$BUILD_DIR/prefix-$ARCH"

    # libplist (no deps)
    build_lib "$BUILD_DIR/libplist-${LIBPLIST_VER}" "$PREFIX" "$ARCH" \
        "--without-cython"

    # libusb (no deps beyond system)
    build_lib "$BUILD_DIR/libusb-${LIBUSB_VER}" "$PREFIX" "$ARCH" \
        "--disable-udev"

    # libimobiledevice-glue (depends on libplist)
    build_lib "$BUILD_DIR/libimobiledevice-glue-${GLUE_VER}" "$PREFIX" "$ARCH" \
        "--without-cython \
         libplist_CFLAGS=-I${PREFIX}/include \
         libplist_LIBS='-L${PREFIX}/lib -lplist-2.0' \
         libplist++_CFLAGS=-I${PREFIX}/include \
         libplist++_LIBS='-L${PREFIX}/lib -lplist++-2.0'"

    # libirecovery (depends on libplist, libusb, libimobiledevice-glue)
    build_lib "$BUILD_DIR/libirecovery-${IRECV_VER}" "$PREFIX" "$ARCH" \
        "--without-cython \
         libplist_CFLAGS=-I${PREFIX}/include \
         libplist_LIBS='-L${PREFIX}/lib -lplist-2.0' \
         libimobiledevice_glue_CFLAGS=-I${PREFIX}/include \
         libimobiledevice_glue_LIBS='-L${PREFIX}/lib -limobiledevice-glue-1.0' \
         libusb_CFLAGS=-I${PREFIX}/include/libusb-1.0 \
         libusb_LIBS='-L${PREFIX}/lib -lusb-1.0'"
done

# Create universal static libs with lipo
echo ""
echo "=== Creating universal static libraries ==="
mkdir -p "$VENDOR_DIR/lib"

for LIB in libplist-2.0.a libplist++-2.0.a libusb-1.0.a libimobiledevice-glue-1.0.a libirecovery-1.0.a; do
    echo "Lipo: $LIB"
    lipo -create \
        "$PREFIX_X86/lib/$LIB" \
        "$PREFIX_ARM/lib/$LIB" \
        -output "$VENDOR_DIR/lib/$LIB"
done

# Copy header
cp /usr/local/include/libirecovery.h "$VENDOR_DIR/include/libirecovery.h"

# Verify
echo ""
echo "=== Verification ==="
for LIB in "$VENDOR_DIR/lib"/*.a; do
    echo "$(basename "$LIB"): $(lipo -info "$LIB")"
done

echo ""
echo "Done! Vendor libs built to $VENDOR_DIR/lib/"
echo "These .a files should be committed to the repository."
```

```swift
import Foundation

// MARK: - C enum bridging helpers

/// Wraps libirecovery C API into Swift-friendly calls.
/// All methods throw `LibirecoveryError` on failure.
final class LibirecoveryBridge {

    // MARK: - Error type

    enum Error: Swift.Error, CustomStringConvertible {
        case noDevice
        case unableToConnect
        case usbUpload
        case usbStatus
        case usbInterface
        case usbConfiguration
        case pipe
        case timeout
        case fileNotFound
        case invalidInput
        case unsupported
        case unknown(irecv_error_t)
        case notConnected

        init(_ code: irecv_error_t) {
            switch code {
            case IRECV_E_NO_DEVICE:         self = .noDevice
            case IRECV_E_UNABLE_TO_CONNECT: self = .unableToConnect
            case IRECV_E_USB_UPLOAD:        self = .usbUpload
            case IRECV_E_USB_STATUS:        self = .usbStatus
            case IRECV_E_USB_INTERFACE:     self = .usbInterface
            case IRECV_E_USB_CONFIGURATION: self = .usbConfiguration
            case IRECV_E_PIPE:              self = .pipe
            case IRECV_E_TIMEOUT:           self = .timeout
            case IRECV_E_FILE_NOT_FOUND:    self = .fileNotFound
            case IRECV_E_INVALID_INPUT:     self = .invalidInput
            case IRECV_E_UNSUPPORTED:       self = .unsupported
            case IRECV_E_SUCCESS:           self = .notConnected // shouldn't happen
            default:                        self = .unknown(code)
            }
        }

        var description: String {
            switch self {
            case .noDevice:         return String(cString: irecv_strerror(IRECV_E_NO_DEVICE))
            case .unableToConnect:  return String(cString: irecv_strerror(IRECV_E_UNABLE_TO_CONNECT))
            case .usbUpload:        return String(cString: irecv_strerror(IRECV_E_USB_UPLOAD))
            case .usbStatus:        return String(cString: irecv_strerror(IRECV_E_USB_STATUS))
            case .usbInterface:     return String(cString: irecv_strerror(IRECV_E_USB_INTERFACE))
            case .usbConfiguration: return String(cString: irecv_strerror(IRECV_E_USB_CONFIGURATION))
            case .pipe:             return String(cString: irecv_strerror(IRECV_E_PIPE))
            case .timeout:          return String(cString: irecv_strerror(IRECV_E_TIMEOUT))
            case .fileNotFound:     return String(cString: irecv_strerror(IRECV_E_FILE_NOT_FOUND))
            case .invalidInput:     return String(cString: irecv_strerror(IRECV_E_INVALID_INPUT))
```

Uses IOKit `IOServiceAddMatchingNotification` to watch for Apple DFU device (VID 0x05AC, PID 0x1227) attach/detach events. Debounces rapid events (200ms) before calling back.

```swift
import Foundation
import IOKit
import IOKit.usb

/// Monitors USB for Apple DFU mode devices (VID 0x05AC, PID 0x1227).
/// Calls delegate on main queue when a device appears or disappears.
protocol USBDeviceMonitorDelegate: AnyObject {
    func dfuDeviceAppeared(serialString: String?)
    func dfuDeviceDisappeared()
}

final class USBDeviceMonitor {
    weak var delegate: USBDeviceMonitorDelegate?

    private var notifyPort: IONotificationPortRef?
    private var matchedIterator: io_iterator_t = 0
    private var terminatedIterator: io_iterator_t = 0
    private var debounceTimer: Timer?

    // Apple DFU device identifiers
    private let appleVendorID = 0x05AC
    private let dfuProductID = 0x1227

    func start() {
        notifyPort = IONotificationPortCreate(kIOMainPortDefault)
        guard let notifyPort = notifyPort else { return }

        // Run notification events on main run loop
        let runLoopSource = IONotificationPortGetRunLoopSource(notifyPort).takeUnretainedValue()
        CFRunLoopAddSource(CFRunLoopGetMain(), runLoopSource, .defaultMode)

        let matchingDict = IOServiceMatching(kIOUSBDeviceClassName) as NSMutableDictionary
        matchingDict[kUSBVendorID] = NSNumber(value: appleVendorID)
        matchingDict[kUSBProductID] = NSNumber(value: dfuProductID)

        let selfPtr = Unmanaged.passUnretained(self).toOpaque()

        // Matched (device appeared)
        let matchResult = IOServiceAddMatchingNotification(
            notifyPort,
            kIOFirstMatchNotification,
            matchingDict,
            { (refCon, iterator) in
                guard let refCon = refCon else { return }
                let monitor = Unmanaged<USBDeviceMonitor>.fromOpaque(refCon).takeUnretainedValue()
                monitor.handleDeviceAppeared(iterator: iterator)
            },
            selfPtr,
            &matchedIterator
        )
        if matchResult == kIOReturnSuccess {
            // Drain existing matches immediately
            handleDeviceAppeared(iterator: matchedIterator)
        }

        // Terminated (device disappeared)
        let termMatching = IOServiceMatching(kIOUSBDeviceClassName) as NSMutableDictionary
        termMatching[kUSBVendorID] = NSNumber(value: appleVendorID)
        termMatching[kUSBProductID] = NSNumber(value: dfuProductID)

        let termResult = IOServiceAddMatchingNotification(
            notifyPort,
            kIOTerminatedNotification,
            termMatching,
            { (refCon, iterator) in
                guard let refCon = refCon else { return }
                let monitor = Unmanaged<USBDeviceMonitor>.fromOpaque(refCon).takeUnretainedValue()
                monitor.handleDeviceDisappeared(iterator: iterator)
            },
            selfPtr,
            &terminatedIterator
        )
        if termResult == kIOReturnSuccess {
            // Drain existing terminations
            handleDeviceDisappeared(iterator: terminatedIterator)
        }
    }

    func stop() {
        debounceTimer?.invalidate()
        debounceTimer = nil
        if matchedIterator != 0 {
            IOObjectRelease(matchedIterator)
            matchedIterator = 0
        }
        if terminatedIterator != 0 {
            IOObjectRelease(terminatedIterator)
            terminatedIterator = 0
        }
        if let port = notifyPort {
            IONotificationPortDestroy(port)
            notifyPort = nil
        }
    }

    private func handleDeviceAppeared(iterator: io_iterator_t) {
        var serial: String?
        while true {
            let device = IOIteratorNext(iterator)
            if device == 0 { break }
            serial = readSerialString(from: device)
            IOObjectRelease(device)
        }

        // Debounce: wait 200ms before notifying
        debounceTimer?.invalidate()
        debounceTimer = Timer.scheduledTimer(withTimeInterval: 0.2, repeats: false) { [weak self] _ in
            DispatchQueue.main.async {
                self?.delegate?.dfuDeviceAppeared(serialString: serial)
            }
        }
    }

    private func handleDeviceDisappeared(iterator: io_iterator_t) {
        // Drain the iterator
        while true {
            let device = IOIteratorNext(iterator)
            if device == 0 { break }
            IOObjectRelease(device)
        }

        debounceTimer?.invalidate()
        debounceTimer = Timer.scheduledTimer(withTimeInterval: 0.2, repeats: false) { [weak self] _ in
            DispatchQueue.main.async {
                self?.delegate?.dfuDeviceDisappeared()
            }
        }
    }

    /// Read the USB serial string from an IOKit device object.
    private func readSerialString(from device: io_object_t) -> String? {
        var serialNumber: String?

        // Get the USB device serial number via IORegistry property
        let properties = IORegistryEntryCreateCFProperty(device, "USB Serial Number" as CFString,
                                                          kCFAllocatorDefault, 0)
        if let props = properties {
            serialNumber = props.takeRetainedValue() as? String
        }

        // Fallback: try "Serial Number" property
        if serialNumber == nil {
            let props2 = IORegistryEntryCreateCFProperty(device, "Serial Number" as CFString,
                                                          kCFAllocatorDefault, 0)
            if let props = props2 {
                serialNumber = props.takeRetainedValue() as? String
            }
        }

        return serialNumber
    }

    deinit {
        stop()
    }
}
```

```swift
import Foundation

/// Represents a recognized device from Devices.plist.
struct DeviceEntry {
    let name: String
    let ibecCodename: String
    let cpid: String   // e.g. "0x8020"
    let bdid: Int      // e.g. 0x0A
}

/// Result of identifying a device from its USB serial string.
enum DeviceIdentification {
    /// Device recognized, PWND status and iBEC codename known.
    case recognized(entry: DeviceEntry, isPWND: Bool, ecid: UInt64?)
    /// CPID is in supported range but BDID is unknown.
    case unsupportedChip(cpid: String)
    /// CPID is known but this specific board (BDID) is not in the config.
    case unknownBoard(cpid: String, bdid: String)
    /// Serial string could not be parsed (missing CPID/BDID fields).
    case unparseable
}

/// Parses DFU device serial strings and looks up device info in Devices.plist.
final class DeviceIdentifier {

    private var entries: [(cpid: String, bdid: Int, entry: DeviceEntry)] = []

    /// Load device entries from a Devices.plist file.
    /// - Parameter url: URL to the plist file.
    /// - Returns: true if loaded successfully.
    @discardableResult
    func load(from url: URL) -> Bool {
        guard let data = try? Data(contentsOf: url),
              let array = try? PropertyListSerialization.propertyList(from: data, options: [], format: nil) as? [[String: Any]] else {
            return false
        }

        entries = array.compactMap { dict in
            guard let cpid = dict["cpid"] as? String,
                  let bdid = dict["bdid"] as? Int,
                  let name = dict["name"] as? String,
                  let codename = dict["ibecCodename"] as? String else {
                return nil
            }
            return (cpid: cpid, bdid: bdid, entry: DeviceEntry(name: name, ibecCodename: codename, cpid: cpid, bdid: bdid))
        }
        return !entries.isEmpty
    }

    /// Supported CPID values (A12 = 0x8020, A13 = 0x8030).
    private let supportedCPIDs: Set<String> = ["0x8020", "0x8030"]

    /// Parse a USB serial string and identify the device.
    /// - Parameter serialString: The device's USB serial string (e.g. "CPID:8020 BDID:0A ... PWND:[usbliter8]").
    /// - Returns: A `DeviceIdentification` result.
    func identify(serialString: String?) -> DeviceIdentification {
        guard let serial = serialString, !serial.isEmpty else {
            return .unparseable
        }

        guard let cpidRaw = serialField(serial, key: "CPID") else {
            return .unparseable
        }
        let cpidKey = "0x\(cpidRaw.lowercased())"

        guard let bdidRaw = serialField(serial, key: "BDID") else {
            return .unparseable
        }

        let isPWND = serial.contains("PWND:[")

        let ecidHex = serialField(serial, key: "ECID")
        let ecid = ecidHex.flatMap { UInt64($0, radix: 16) }

        // Look up in entries
        let bdidInt = Int(bdidRaw, radix: 16) ?? Int(bdidRaw) ?? -1
        if let match = entries.first(where: { $0.cpid == cpidKey && $0.bdid == bdidInt }) {
            return .recognized(entry: match.entry, isPWND: isPWND, ecid: ecid)
        }

        // Check if chip is supported at all
        if supportedCPIDs.contains(cpidKey) {
            return .unknownBoard(cpid: cpidKey, bdid: bdidRaw)
        }

        return .unsupportedChip(cpid: cpidKey)
    }

    /// Extract a field value from the serial string.
    /// Serial format: "CPID:8020 BDID:0A ECID:000000... PWND:[usbliter8]"
    private func serialField(_ serial: String, key: String) -> String? {
        for part in serial.split(separator: " ") {
            if part.hasPrefix("\(key):") {
                return String(part.split(separator: ":", maxSplits: 1).last ?? "")
            }
        }
        return nil
    }
}
```

```swift
import Foundation

/// Resolves the path to a patched iBEC file for a given codename.
/// Priority: user override directory → app bundle Resources.
final class IBECResolver {

    /// User override directory (~/Library/Application Support/EraseA12/boot/)
    static let userOverrideDir = FileManager.default.homeDirectoryForCurrentUser
        .appendingPathComponent("Library/Application Support/EraseA12/boot", isDirectory: true)

    /// App bundle Resources/boot/ directory
    private let bundleBootDir: URL?

    /// iBEC filename template: iBEC.<codename>.RELEASE.patched
    static func ibecFilename(codename: String) -> String {
        return "iBEC.\(codename).RELEASE.patched"
    }

    init(bundle: Bundle = .main) {
        bundleBootDir = bundle.url(forResource: "boot", withExtension: nil)
    }

    /// Resolve the iBEC file URL for the given codename.
    /// Returns nil if no file is found at either location.
    func resolve(codename: String) -> URL? {
        let filename = Self.ibecFilename(codename: codename)

        // Priority 1: user override
        let userPath = Self.userOverrideDir.appendingPathComponent(filename)
        if FileManager.default.fileExists(atPath: userPath.path) {
            return userPath
        }

        // Priority 2: app bundle
        if let bundleDir = bundleBootDir {
            let bundlePath = bundleDir.appendingPathComponent(filename)
            if FileManager.default.fileExists(atPath: bundlePath.path) {
                return bundlePath
            }
        }

        return nil
    }

    /// Check whether an iBEC file exists for the given codename.
    func hasIBEC(codename: String) -> Bool {
        return resolve(codename: codename) != nil
    }
}
```

```swift
    }

    /// Resolve the iBEC file URL for the given codename.
    /// Returns nil if no file is found at either location.
    func resolve(codename: String) -> URL? {
        let filename = Self.ibecFilename(codename: codename)

        // Priority 1: user override
        let userPath = effectiveUserOverrideDir.appendingPathComponent(filename)
        if FileManager.default.fileExists(atPath: userPath.path) {
            return userPath
        }

        // Priority 2: app bundle
        if let bundleDir = bundleBootDir {
            let bundlePath = bundleDir.appendingPathComponent(filename)
            if FileManager.default.fileExists(atPath: bundlePath.path) {
                return bundlePath
            }
        }

        return nil
    }
```

```swift
// MARK: - State machine

/// States of the obliteration flow.
enum ObliterationState: Equatable {
    case idle
    case uploading
    case booting
    case waitingRecovery
    case sendingCommands
    case rebooting
    case done
    case failed(ObliterationError)
}

/// Progress callback type.
typealias ObliterationProgressHandler = (ObliterationState) -> Void

/// Drives the complete obliteration flow:
/// upload iBEC → boot → wait for Recovery → send NVRAM commands → reboot
final class ObliterationEngine {

    private let bridge: LibirecoveryProtocol
    private let ibecResolver: IBECResolver
    private let deviceIdentifier: DeviceIdentifier
    private let workQueue = DispatchQueue(label: "com.prdgmshift.erasea12.obliteration", qos: .userInitiated)

    var onProgress: ObliterationProgressHandler?

    init(bridge: LibirecoveryProtocol,
         ibecResolver: IBECResolver,
         deviceIdentifier: DeviceIdentifier) {
        self.bridge = bridge
        self.ibecResolver = ibecResolver
        self.deviceIdentifier = deviceIdentifier
    }

    /// Execute the obliteration flow for a device identified by its serial string.
    /// - Parameters:
    ///   - serialString: The USB serial string from the DFU device.
    ///   - completion: Called on main queue with final state (.done or .failed).
    func execute(serialString: String, completion: @escaping (ObliterationState) -> Void) {
        workQueue.async { [weak self] in
            guard let self = self else { return }
            let result = self._execute(serialString: serialString)
            DispatchQueue.main.async {
                completion(result)
            }
        }
    }

    private func _execute(serialString: String) -> ObliterationState {
        // 1. Identify device
        let identification = deviceIdentifier.identify(serialString: serialString)

        guard case .recognized(let entry, let isPWND, let ecid) = identification else {
            switch identification {
            case .unsupportedChip(let cpid):
                return .failed(.unsupportedDevice(cpid: cpid, bdid: nil))
            case .unknownBoard(let cpid, let bdid):
                return .failed(.unsupportedDevice(cpid: cpid, bdid: bdid))
            case .unparseable:
                return .failed(.noDevice)
            default:
                return .failed(.unknown(underlying: "Unexpected identification result"))
            }
        }

        guard isPWND else {
            return .failed(.notPWND)
        }

        // 2. Resolve iBEC file
        guard let ibecURL = ibecResolver.resolve(codename: entry.ibecCodename) else {
            return .failed(.ibecNotFound(codename: entry.ibecCodename))
        }

        // 3. Connect to device
        do {
            emit(.uploading)
            try bridge.connect(ecid: ecid ?? 0)

            // 4. Upload iBEC
            try bridge.sendFile(at: ibecURL.path)
        } catch let error as LibirecoveryBridge.Error {
            return .failed(.uploadFailed(underlying: error.description))
        } catch {
            return .failed(.uploadFailed(underlying: error.localizedDescription))
        }

        // 5. Boot iBEC
        do {
            emit(.booting)
            try bridge.resetCounters()
            try bridge.finishTransfer()
            // Send CUSTOM_BOOT equivalent via send_command
            // In DFU mode, we need to use USB control transfers directly
            // libirecovery handles this internally when we call the right sequence
        } catch {
            // Boot step may partially fail but device can still transition
            // to recovery mode — we continue waiting
        }

        // 6. Disconnect from DFU and wait for Recovery mode
        bridge.disconnect()
        emit(.waitingRecovery)

        // Wait up to 60 seconds for device to reappear in Recovery mode
        let deadline = Date().addingTimeInterval(60)
        var connected = false
        while Date() < deadline {
            Thread.sleep(forTimeInterval: 1.0)
            do {
                try bridge.connect(ecid: ecid ?? 0)
                let mode = bridge.getMode()
                // Recovery mode values: 0x1280-0x1283
                if mode >= 0x1280 && mode <= 0x1283 {
                    connected = true
                    break
                }
                bridge.disconnect()
            } catch {
                // Not yet in recovery, keep waiting
            }
        }

        guard connected else {
            return .failed(.recoveryTimeout)
        }

        // 7. Send obliteration commands
        emit(.sendingCommands)
        do {
            try bridge.setenv("oblit-inprogress", "5")
            try bridge.setenv("auto-boot", "true")
            try bridge.saveenv()
        } catch {
            return .failed(.commandFailed(underlying: error.localizedDescription))
        }

        // 8. Reboot
        emit(.rebooting)
        do {
            try bridge.reboot()
        } catch {
            // Reboot may throw but device still reboots — ignore, matching Python behavior
        }

        bridge.disconnect()
        emit(.done)
        return .done
    }

    private func emit(_ state: ObliterationState) {
        guard let handler = onProgress else { return }
        DispatchQueue.main.async {
            handler(state)
        }
    }
}
```

```swift
/// A horizontal 4-step indicator showing progress through the obliteration wizard.
/// Displays 4 dots/segments connected by lines, with the current step highlighted.
class StepIndicatorView: NSView {

    enum Step: Int, CaseIterable {
        case waiting = 0
        case confirm = 1
        case executing = 2
        case done = 3

        var localizedTitle: String {
            switch self {

            case .confirm:   return NSLocalizedString("step.confirm", value: "Confirm", comment: "Step 2 title")

            case .done:      return NSLocalizedString("step.done", value: "Done", comment: "Step 4 title")
            }
        }
    }

    var currentStep: Step = .waiting {
        didSet { needsDisplay = true }
    }

    private let dotRadius: CGFloat = 6.0
    private let lineLength: CGFloat = 40.0
    private let accentColor = NSColor.systemBlue
    private let inactiveColor = NSColor(white: 0.4, alpha: 0.6)
    private let completedColor = NSColor.systemGreen

    override init(frame frameRect: NSRect) {
        super.init(frame: frameRect)
        intrinsicContentSize = NSSize(width: 240, height: 40)
    }

    required init?(coder: NSCoder) {
        super.init(coder: coder)
        intrinsicContentSize = NSSize(width: 240, height: 40)
    }

    override var intrinsicContentSize: NSSize {
        return NSSize(width: 240, height: 40)
    }

    override func draw(_ dirtyRect: NSRect) {
        guard let context = NSGraphicsContext.current?.cgContext else { return }

        let steps = Step.allCases
        let totalWidth = CGFloat(steps.count - 1) * lineLength
        let startX = (bounds.width - totalWidth) / 2.0
        let centerY = bounds.height / 2.0 - 8

        for (index, step) in steps.enumerated() {
            let x = startX + CGFloat(index) * lineLength

            // Draw connecting line
            if index > 0 {
                let prevX = startX + CGFloat(index - 1) * lineLength
                let lineColor = index <= currentStep.rawValue ? completedColor : inactiveColor
                context.setStrokeColor(lineColor.cgColor)
                context.setLineWidth(2.0)
                context.move(to: CGPoint(x: prevX + dotRadius + 2, y: centerY))
                context.addLine(to: CGPoint(x: x - dotRadius - 2, y: centerY))
                context.strokePath()
            }

            // Draw dot
            let dotColor: NSColor
            if index < currentStep.rawValue {
                dotColor = completedColor
            } else if index == currentStep.rawValue {
                dotColor = accentColor
            } else {
                dotColor = inactiveColor
            }

            context.setFillColor(dotColor.cgColor)
            let rect = CGRect(x: x - dotRadius, y: centerY - dotRadius,
                              width: dotRadius * 2, height: dotRadius * 2)
            context.fillEllipse(in: rect)

            // Draw step title
            let titleColor: NSColor = index <= currentStep.rawValue ? .textColor : inactiveColor
            let attrs: [NSAttributedString.Key: Any] = [
                .font: NSFont.systemFont(ofSize: 10, weight: .medium),
                .foregroundColor: titleColor,
            ]
            let title = step.localizedTitle
            let titleSize = title.size(withAttributes: attrs)
            let titlePoint = CGPoint(x: x - titleSize.width / 2, y: centerY - dotRadius - 16)
            title.draw(at: titlePoint, withAttributes: attrs)
        }
    }
```

```swift
import AppKit

/// Main window controller that orchestrates the 4-step wizard flow.
class MainWindowController: NSWindowController {

    private let stepIndicator = StepIndicatorView()
    private let glassBackground = GlassBackgroundView()

    private let waitingVC = WaitingStepViewController()
    private let confirmVC = ConfirmStepViewController()
    private let executingVC = ExecutingStepViewController()
    private let doneVC = DoneStepViewController()

    // Core components
    private let deviceMonitor = USBDeviceMonitor()
    private let deviceIdentifier = DeviceIdentifier()
    private let ibecResolver = IBECResolver()
    private var obliterationEngine: ObliterationEngine?

    // Current device state
    private var currentSerialString: String?
    private var currentIdentification: DeviceIdentification?

    convenience init() {
        let window = NSWindow(
            contentRect: NSRect(x: 0, y: 0, width: 520, height: 400),
            styleMask: [.titled, .closable, .fullSizeContentView],
            backing: .buffered,
            defer: false
        )
        window.title = "EraseA12"
        window.titlebarAppearsTransparent = true
        window.titleVisibility = .hidden
        window.isMovableByWindowBackground = true
        window.center()
        window.minSize = NSSize(width: 520, height: 400)
        window.maxSize = NSSize(width: 520, height: 400)

        self.init(window: window)
        setupUI()
        setupCore()
    }

    private func setupUI() {
        guard let contentView = window?.contentView else { return }

        // Glass background
        glassBackground.autoresizingMask = [.width, .height]
        glassBackground.frame = contentView.bounds
        contentView.addSubview(glassBackground, positioned: .below, relativeTo: nil)

        // Step indicator at top
        stepIndicator.currentStep = .waiting
        stepIndicator.translatesAutoresizingMaskIntoConstraints = false
        contentView.addSubview(stepIndicator)

        NSLayoutConstraint.activate([
            stepIndicator.topAnchor.constraint(equalTo: contentView.topAnchor, constant: 20),
            stepIndicator.centerXAnchor.constraint(equalTo: contentView.centerXAnchor),
            stepIndicator.widthAnchor.constraint(equalToConstant: 240),
            stepIndicator.heightAnchor.constraint(equalToConstant: 40),
        ])

        // Container for step view controllers
        let container = NSView()
        container.translatesAutoresizingMaskIntoConstraints = false
        contentView.addSubview(container)

        NSLayoutConstraint.activate([
            container.topAnchor.constraint(equalTo: stepIndicator.bottomAnchor, constant: 8),
            container.leadingAnchor.constraint(equalTo: contentView.leadingAnchor),
            container.trailingAnchor.constraint(equalTo: contentView.trailingAnchor),
            container.bottomAnchor.constraint(equalTo: contentView.bottomAnchor),
        ])

        // Start with waiting step
        switchToStep(.waiting)
    }

    private func setupCore() {
        deviceMonitor.delegate = self

        // Load Devices.plist from bundle
        if let plistURL = Bundle.main.url(forResource: "Devices", withExtension: "plist") {
            deviceIdentifier.load(from: plistURL)
        }

        // Wire up confirm step
        confirmVC.onConfirm = { [weak self] in
            self?.startObliteration()
        }

        // Wire up done step
        doneVC.onEraseAnother = { [weak self] in
            self?.resetToWaiting()
        }
    }

    // MARK: - Step switching

    private enum WizardStep {
        case waiting, confirm, executing, done
    }

    private func switchToStep(_ step: WizardStep) {
        guard let contentView = window?.contentView else { return }

        // Remove current child VC view
        for vc in children {
            vc.view.removeFromSuperview()
            vc.removeFromParent()
        }

        let vc: NSViewController
        let indicatorStep: StepIndicatorView.Step

        switch step {
        case .waiting:
            vc = waitingVC
            indicatorStep = .waiting
        case .confirm:
            vc = confirmVC
            indicatorStep = .confirm
        case .executing:
            vc = executingVC
            indicatorStep = .executing
        case .done:
            vc = doneVC
            indicatorStep = .done
        }

        addChild(vc)
        vc.view.frame = contentView.bounds.insetBy(dx: 0, dy: 60)
        vc.view.autoresizingMask = [.width, .height]
        contentView.addSubview(vc.view, positioned: .below, relativeTo: stepIndicator)
        stepIndicator.currentStep = indicatorStep
    }

    // MARK: - Obliteration

    private func startObliteration() {
        guard let serial = currentSerialString else { return }

        let bridge = LibirecoveryBridge()
        obliterationEngine = ObliterationEngine(
            bridge: bridge,
            ibecResolver: ibecResolver,
            deviceIdentifier: deviceIdentifier
        )

        switchToStep(.executing)

        obliterationEngine?.onProgress = { [weak self] state in
            self?.executingVC.updatePhase(state)
        }

        obliterationEngine?.execute(serialString: serial) { [weak self] finalState in
            guard let self = self else { return }
            self.switchToStep(.done)
            switch finalState {
            case .done:
                self.doneVC.configureSuccess()
            case .failed(let error):
                self.doneVC.configureFailure(error: error)
            default:
                break
            }
        }
    }

    @objc func resetToWaiting() {
        currentSerialString = nil
        currentIdentification = nil
        obliterationEngine = nil
        switchToStep(.waiting)
    }
}

// MARK: - USBDeviceMonitorDelegate

extension MainWindowController: USBDeviceMonitorDelegate {
    func dfuDeviceAppeared(serialString: String?) {
        currentSerialString = serialString
        let identification = deviceIdentifier.identify(serialString: serialString)
        currentIdentification = identification

        switch identification {
        case .recognized(let entry, let isPWND, _):
            let canErase = isPWND && ibecResolver.hasIBEC(codename: entry.ibecCodename)
            confirmVC.configure(
                deviceName: entry.name,
                cpid: entry.cpid,
                bdid: String(format: "0x%02X", entry.bdid),
                isPWND: isPWND,
                ibecCodename: entry.ibecCodename,
                canErase: canErase
            )
            switchToStep(.confirm)

        case .unsupportedChip, .unknownBoard, .unparseable:
            // Stay on waiting step but update message
            break
        }
    }

    func dfuDeviceDisappeared() {
        currentSerialString = nil
        currentIdentification = nil
        switchToStep(.waiting)
    }
}
```

```bash
#!/bin/bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
BUILD_DIR="$PROJECT_DIR/build/Release"
DMG_NAME="EraseA12-1.0.0"
DMG_DIR="$PROJECT_DIR/build/dmg"

echo "=== EraseA12 DMG Packaging ==="

# Build release
echo "Building Release..."
cd "$PROJECT_DIR"
xcodegen generate
xcodebuild -project EraseA12.xcodeproj -scheme EraseA12 -configuration Release \
    CONFIGURATION_BUILD_DIR="$BUILD_DIR" \
    clean build

# Ad-hoc sign
echo "Ad-hoc signing..."
codesign --force --deep --sign - "$BUILD_DIR/EraseA12.app"

# Create DMG structure
echo "Creating DMG..."
rm -rf "$DMG_DIR"
mkdir -p "$DMG_DIR"
cp -R "$BUILD_DIR/EraseA12.app" "$DMG_DIR/"

# Create a README shortcut
cat > "$DMG_DIR/README-打开方式.txt" << 'EOF'
首次打开方法：
1. 右键点击 EraseA12.app → 选择"打开"
2. 或在"系统设置 → 隐私与安全性"中点击"仍要打开"

How to open for the first time:
1. Right-click EraseA12.app → Select "Open"
2. Or go to "System Settings → Privacy & Security" and click "Open Anyway"
EOF

# Create DMG
hdiutil create -volname "EraseA12" \
    -srcfolder "$DMG_DIR" \
    -ov -format UDZO \
    "${DMG_NAME}.dmg"

echo "Done: ${DMG_NAME}.dmg"
```
